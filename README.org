* Primera clase de programación

Programar la declinación.

#+begin_src C
  #include <stdio.h>
#include <stdlib.h>
#include <math.h>

#define PI         3.1415926535897932384626433832795028841971

double r2d(double radianes);
double d2r(double grados);
double delta(int N);

int main(int argc, char *argv[])
{
  int N;
  double d;

  N = 355;
  d = delta(N);

  printf("%f\n" , d);
  
  return 0;
}

/* Funciones basicas */
/* ---------------------------------------------------------------------- */

/* radianes a grados */
double r2d(double radianes){
  return (180.0/PI)*radianes;
}

/* grados a radianes */
double d2r(double grados){
  return (PI/180.0)*grados;
}

/* Declinaci'on */
double delta(int N){
  double decl;
/* cooper */
    decl = 23.45 * sin(2.0*PI*(284 + N)/365.0);

  return decl;
}
#+end_src
